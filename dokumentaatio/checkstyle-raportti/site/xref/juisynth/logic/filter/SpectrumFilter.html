6   package juisynth.logic.filter;
7   
8   import juisynth.logic.Waveform;
9   import juisynth.logic.WaveformCalculator;
10  
11  /**
12   * A naive implementation of a filter, attenuating the harmonics of a signal,
13   * strengthening the fundamental frequency.
14   * @author juicyp
15   */
16  public class SpectrumFilter extends Filter {
17      
18      /**
19       * Sets the given FilterSettings object into the field containing object of said type.
20       * @param settings FilterSettings object containing information relevant to filters.
21       */
22      public SpectrumFilter(FilterSettings settings) {
23          this.settings = settings;
24      }
25  
26      @Override
27      public double generateFilter(double phase, double amplitude) {
28          return (1 - settings.getDepth()) * amplitude + settings.getDepth() * WaveformCalculator.calculateWaveformY(phase, Waveform.SIN);
29      }
30  }
