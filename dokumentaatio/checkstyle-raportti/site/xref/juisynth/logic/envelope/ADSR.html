6   package juisynth.logic.envelope;
7   
8   import juisynth.logic.player.Settings;
9   import juisynth.logic.signal.SignalStatus;
10  
11  /**
12   *
13   * @author juicyp
14   */
15  public class ADSR extends EnvelopeGenerator {
16  
17      // Instantiating fields at declaration here as well for uniformity
18      private int attack = 0;
19      private int decay = 0;
20      private int sustain = 100;
21      private int release = 0;
22  
23      private double noteOffLevel;
24      // Watch out for unexpected behaviour due to overflowing buffer indices
25      private int noteOffBufferIndex = -1;
26  
27      public ADSR() {
28      }
29  
30      public void setAttack(int attack) {
31          if (attack < 0) {
32              return;
33          }
34          this.attack = attack;
35      }
36  
37      public void setDecay(int decay) {
38          if (decay < 0) {
39              return;
40          }
41          this.decay = decay;
42      }
43  
44      public void setSustain(int sustain) {
45          if (sustain < 0 || sustain > 100) {
46              return;
47          }
48          this.sustain = sustain;
49      }
50  
51      public void setRelease(int release) {
52          if (release < 0) {
53              return;
54          }
55          this.release = release;
56      }
57  
58      // Useless casts?
59      @Override
60      public double generateEnvelope(SignalStatus signal) {
61          int elapsedTime = (int) (signal.getBufferIndex() / (double) Settings.SAMPLE_RATE * 1000);
62          double amplitude = 0;
63  
64          if (signal.getActiveNote()) {
65              noteOffBufferIndex = -1;
66              if (attack >= elapsedTime) {
67                  amplitude = elapsedTime / (double) attack;
68              } else if (attack + decay >= elapsedTime) {
69                  amplitude = 1 - (1 - sustain) * ((elapsedTime - attack) / (double) decay);
70              } else {
71                  amplitude = sustain / (double) 100;
72              }
73              noteOffLevel = amplitude;
74              return amplitude;
75          }
76  
77          if (noteOffBufferIndex == -1) {
78              noteOffBufferIndex = signal.getBufferIndex();
79          }
80  
81          int noteOffTime = (int) (noteOffBufferIndex / (double) Settings.SAMPLE_RATE * 1000);
82          // Time elapsed after noteOff isn't longer than release
83          if (release >= elapsedTime - noteOffTime) {
84              amplitude = noteOffLevel - noteOffLevel * ((elapsedTime - noteOffTime) / (double) release);
85          }
86  
87          return amplitude;
88      }
89  
90  }
