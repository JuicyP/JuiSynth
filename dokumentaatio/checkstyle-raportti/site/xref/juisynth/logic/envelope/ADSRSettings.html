6   package juisynth.logic.envelope;
7   
8   /**
9    * Contains settings specific to an ADSR-envelope generator.
10   * @author juicyp
11   */
12  public class ADSRSettings extends EnvelopeSettings {
13      
14      private int attack = 0;
15      private int decay = 0;
16      private double sustain = 1;
17      private int release = 0;
18      
19      public int getAttack() {
20          return attack;
21      }
22  
23      /**
24       * Sets non negative value for the attack, specified in milliseconds.
25       * @param attack The time value of the attack phase.
26       */
27      public void setAttack(int attack) {
28          if (attack < 0) {
29              return;
30          }
31          this.attack = attack;
32      }
33      
34      public int getDecay() {
35          return decay;
36      }
37  
38      /**
39       * Sets non negative value for the decay, specified in milliseconds.
40       * @param decay The time value of the decay phase.
41       */
42      public void setDecay(int decay) {
43          if (decay < 0) {
44              return;
45          }
46          this.decay = decay;
47      }
48      
49      public double getSustain() {
50          return sustain;
51      }
52  
53      /**
54       * Sets the amplitude for the generated envelope during the sustain phase.
55       * Value must be between 0 and 1 inclusive.
56       * @param sustain The amplitude of the sustain phase.
57       */
58      public void setSustain(double sustain) {
59          if (sustain < 0 || sustain > 1) {
60              return;
61          }
62          this.sustain = sustain;
63      }
64      
65      public int getRelease() {
66          return release;
67      }
68  
69      /**
70       * Sets non negative value for the release, specified in milliseconds.
71       * @param release The time value of the release phase.
72       */
73      public void setRelease(int release) {
74          if (release < 0) {
75              return;
76          }
77          this.release = release;
78      }
79  }
