6   package juisynth.logic.player;
7   
8   import juisynth.logic.signal.SignalStatus;
9   import juisynth.logic.signal.SignalSource;
10  import java.util.Arrays;
11  import javax.sound.sampled.AudioFormat;
12  import javax.sound.sampled.AudioSystem;
13  import javax.sound.sampled.DataLine;
14  import javax.sound.sampled.LineUnavailableException;
15  import javax.sound.sampled.SourceDataLine;
16  import javax.swing.SwingWorker;
17  
18  /**
19   * Starts a thread and writes sample data fetched from SignalSource into
20   * SourceDataLine.
21   *
22   * @author juicyp
23   */
24  public class Player {
25  
26      private AudioFormat format;
27      private DataLine.Info info;
28      private SourceDataLine audioline;
29  
30      private SwingWorker worker;
31  
32      private byte[] sampleBuffer = new byte[Settings.BUFFER_SIZE];
33      private SignalSource signalSource;
34      private int bufferIndex;
35      private double frequency = 440;
36      private double amp = 1;
37  
38      /**
39       * Constructor for Player sets format according to values specified in
40       * fields.
41       */
42      public Player() {
43          // The constuctor of AudioFormat takes the sample rate, sample resolution in bits,
44          // amount of channels, signedness (true for signed),
45          // endianness (true for big endian) in that order as format information.
46          format = new AudioFormat(Settings.SAMPLE_RATE, Settings.SAMPLE_SIZE,
47                  Settings.CHANNELS, Settings.SIGNED, Settings.BIG_ENDIAN);
48          info = new DataLine.Info(SourceDataLine.class, format);
49      }
50  
51      public double getAmp() {
52          return amp;
53      }
54  
55      public void setAmp(double amp) {
56          if (amp < 0 || amp > 1) {
57              return;
58          }
59          this.amp = amp;
60      }
61  
62      public double getFrequency() {
63          return frequency;
64      }
65  
66      public void setFrequency(double frequency) {
67          if (frequency < 0) {
68              return;
69          }
70          this.frequency = frequency;
71      }
72  
73      public void setSignalSource(SignalSource signalSource) {
74          this.signalSource = signalSource;
75      }
76  
77      /**
78       * Starts the player on a new thread.
79       *
80       * @see stopPlayer
81       */
82      public void startPlayer() {
83          if (signalSource != null && worker == null) {
84              worker = new SwingWorker<Void, Void>() {
85  
86                  private boolean done;
87  
88                  @Override
89                  protected Void doInBackground() {
90  
91                      done = false;
92  
93                      try {
94                          audioline = (SourceDataLine) AudioSystem.getLine(info);
95                          audioline.open(format);
96                          audioline.start();
97                          bufferIndex = 0;
98  
99                          while (!done) {
100                             writeBuffer();
101                         }
102 
103                     } catch (LineUnavailableException e) {
104 
105                     } finally {
106                         audioline.drain();
107                         audioline.close();
108                     }
109                     return null;
110                 }
111 
112                 @Override
113                 protected void done() {
114                     done = true;
115                 }
116             };
117             worker.execute();
118         }
119     }
120 
121     /**
122      * Sets a flag to kill the thread.
123      *
124      * @see startPlayer
125      */
126     public void stopPlayer() {
127         if (worker != null) {
128             worker.cancel(true);
129             worker = null;
130         }
131     }
132 
133     private byte[] writeBuffer() {
134 
135         Arrays.fill(sampleBuffer, (byte) 0);
136 
137         int index = 0;
138 
139         SignalStatus signal = new SignalStatus(bufferIndex, frequency);
140 
141         for (int i = 0; i < Settings.SAMPLES_PER_BUFFER; i++) {
142             // Maybe just use same SignalStatus instance for successive sample fetches?
143             // Less memory garbage
144             signal.setAmplitude(0);
145             signal.setBufferIndex(bufferIndex++);
146             signal.setFrequency(frequency);
147             signal.setCompletePeriod(false);
148             signal.setActiveOperatorCount(0);
149 
150             signalSource.generateSample(signal);
151             signal.setAmplitude(signal.getAmplitude() * amp);
152 
153             double ds = signal.getAmplitude() * Short.MAX_VALUE;
154             short ss = (short) Math.round(ds);
155             // Big endian, shift first eight bits and add as first part of sample
156             sampleBuffer[index++] = (byte) (ss >> 8);
157             sampleBuffer[index++] = (byte) (ss & 0xFF);
158         }
159         audioline.write(sampleBuffer, 0, Settings.BUFFER_SIZE);
160         return sampleBuffer;
161     }
162 }
