6   package juisynth.gui;
7   
8   import juisynth.logic.player.Player;
9   import juisynth.logic.oscillator.Oscillator;
10  import java.awt.Container;
11  import java.awt.Dimension;
12  import javax.swing.*;
13  
14  /**
15   * Main UI implementation of JuiSynth.
16   * @author juicyp
17   */
18  public class GUI implements Runnable {
19      
20      private JFrame frame;
21      
22      private Oscillator oscillator1;
23      private Oscillator oscillator2;
24      private Oscillator oscillator3;
25      private Oscillator oscillator4;
26             
27      private Player player;
28  
29      @Override
30      public void run() {
31          oscillator1 = new Oscillator();
32          oscillator1.setAdd(true);
33          
34          oscillator2 = new Oscillator();
35          oscillator2.setBypass(true);
36          oscillator2.setSignalSource(oscillator1);
37          
38          oscillator3 = new Oscillator();
39          oscillator3.setBypass(true);
40          oscillator3.setSignalSource(oscillator2);
41          
42          oscillator4 = new Oscillator();
43          oscillator4.setBypass(true);
44          oscillator4.setSignalSource(oscillator3);
45          
46          player = new Player();
47          player.setSignalSource(oscillator4);
48          
49          frame = new JFrame("JuiSynth");
50          frame.setPreferredSize(new Dimension(1200, 400));
51          
52          frame.setDefaultCloseOperation(WindowConstants.EXIT_ON_CLOSE);
53          
54          instantiateComponents(frame.getContentPane());
55          
56          frame.pack();
57          frame.setVisible(true);
58      }
59      
60      private void instantiateComponents(Container container) {
61          BoxLayout layout = new BoxLayout(container, BoxLayout.X_AXIS);
62          container.setLayout(layout);
63          
64          container.add(new OperatorPanel(oscillator1));
65          container.add(new OperatorPanel(oscillator2));       
66          container.add(new OperatorPanel(oscillator3));
67          container.add(new OperatorPanel(oscillator4));
68          container.add(new PlayerPanel(player));
69      }
70      
71      public JFrame getFrame() {
72          return frame;
73      }
74  }
