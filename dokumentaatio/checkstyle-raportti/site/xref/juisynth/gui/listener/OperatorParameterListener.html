6   package juisynth.gui.listener;
7   
8   import juisynth.gui.OperatorParameter;
9   import javax.swing.JSlider;
10  import javax.swing.event.ChangeEvent;
11  import javax.swing.event.ChangeListener;
12  import juisynth.logic.Patch;
13  
14  /**
15   * Defines a listener for the parameters as defined by the implementation of Operator.
16   * @author juicyp
17   */
18  public class OperatorParameterListener implements ChangeListener {
19      
20      private Patch patch;
21      private JSlider slider;
22      private OperatorParameter operatorParameter;
23  
24      /**
25       * Constructor for the listener, supplying the slider, parameter enumerator and the
26       * Patch component to be connected to.
27       * @param patch The Patch object.
28       * @param slider The slider relating to a parameter.
29       * @param operatorParameter The parameter.
30       */
31      public OperatorParameterListener(Patch patch, JSlider slider, OperatorParameter operatorParameter) {
32          this.patch = patch;
33          this.slider = slider;
34          this.operatorParameter = operatorParameter;
35      }
36  
37      @Override
38      public void stateChanged(ChangeEvent ce) {
39          switch (operatorParameter) {
40              case AMP:
41                  patch.setAmp(slider.getValue() / (double) 100);
42                  break;
43              case FM:
44                  patch.setFmDepth(slider.getValue() / (double) 100);
45                  break;
46              case AM:
47                  patch.setAmDepth(slider.getValue() / (double) 100);
48                  break;
49              default:
50                  break;
51          }
52      }
53  }
