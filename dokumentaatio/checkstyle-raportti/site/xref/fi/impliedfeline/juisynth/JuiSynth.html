6   package fi.impliedfeline.juisynth;

8   import fi.impliedfeline.juisynth.logic.*;
9   import javax.sound.sampled.AudioFormat;
10  import javax.sound.sampled.AudioSystem;
11  import javax.sound.sampled.DataLine;
12  import javax.sound.sampled.SourceDataLine;
13  import javax.sound.sampled.LineUnavailableException;
14  import java.util.Arrays;
15  
16  /**
17   *
18   * @author juicyp
19   */
20  public class JuiSynth {
21  
22      public static void main(String[] args) {
23  
24          // BEWARE! Software is LOUD
25          // Prototype modeled after Dr. Dobb's synth articles 
26          // http://www.drdobbs.com/jvm/creating-music-components-in-java/229700113?pgno=1
27          // Testing
28          int sampleRate = 44100; // 44.1kHz is a sample rate with fairly high fidelity
29  
30          // The constuctor of AudioFormat takes the sample rate, sample resolution in bits,
31          // amount of channels, signedness (true for signed),
32          // endianness (true for big endian) in that order as format information.
33          AudioFormat format = new AudioFormat(sampleRate, 16, 1, true, true);
34          DataLine.Info info = new DataLine.Info(SourceDataLine.class, format);
35          SourceDataLine audioline = null;
36  
37          // Larger buffer less suspectible to system running slow
38          // Working on a laptop from the last decade
39          int bufferSize = 10000;
40          // 16-bit samples so two indices of a byte array represent a single sample.
41          // Thus, amount of samples per buffer is half the size of the array.
42          int samplesPerBuffer = bufferSize / 2;
43          byte[] sampleBuffer = new byte[bufferSize];
44          Arrays.fill(sampleBuffer, (byte) 0);
45  
46          try {
47  
48              audioline = (SourceDataLine) AudioSystem.getLine(info);
49              audioline.open(format);
50              audioline.start();
51  
52              // Fetching system time on each iteration too slow,
53              // distorted signal. Reducing a number on each iteration instead,
54              // writing the amount of buffers on variable
55              int bufferCount = 10;
56  
57              while (bufferCount > 0) {
58                  generateWaveIntoBuffer(sampleBuffer, sampleRate, 440, samplesPerBuffer);
59                  audioline.write(sampleBuffer, 0, bufferSize);
60                  bufferCount -= 1;
61              }
62  
63          } catch (LineUnavailableException e) {
64  
65          } finally {
66  
67              audioline.drain();
68              audioline.close();
69  
70          }
71      }
72  
73      // Calculate the y value of a sine wave at frequency for each discrete point x,
74      // and put into buffer.
75      public static void generateWaveIntoBuffer(byte[] sampleBuffer, int sampleRate, double frequency, int samplesPerBuffer) {
76  
77          Oscillator oscillator = new Oscillator();
78          oscillator.setWaveform(Oscillator.Waveform.NOI);
79          oscillator.setAdd(true);
80          int index = 0;
81  
82          for (int i = 0; i < samplesPerBuffer; i++) {
83  
84              // Maybe just use same SignalStatus instance for successive sample fetches?
85              // Less memory garbage
86              SignalStatus signal = new SignalStatus(sampleRate, i, frequency);
87              oscillator.generateSample(signal);
88  
89              double ds = signal.getAmplitude() * Short.MAX_VALUE;
90              short ss = (short) Math.round(ds);
91              // Big endian, shift first eight bits and add as first part of sample
92              sampleBuffer[index++] = (byte) (ss >> 8);
93              sampleBuffer[index++] = (byte) (ss & 0xFF);
94          }
95      }
