6   package fi.impliedfeline.juisynth;
7   
8   import javax.sound.sampled.AudioFormat;
9   import javax.sound.sampled.AudioSystem;
10  import javax.sound.sampled.DataLine;
11  import javax.sound.sampled.SourceDataLine;
12  import javax.sound.sampled.LineUnavailableException;
13  import java.util.Arrays;
14  
15  /**
16   *
17   * @author juicyp
18   */
19  public class JuiSynth {
20  
21      public static void main(String[] args) {
22  
23          // Prototype modeled after Dr. Dobb's synth articles 
24          // http://www.drdobbs.com/jvm/creating-music-components-in-java/229700113?pgno=1
25          // TODO: Refactor everything
26          int sampleRate = 44100; // 44.1kHz is a sample rate with fairly high fidelity
27  
28          // The constuctor of AudioFormat takes the sample rate, sample resolution in bits,
29          // amount of channels, signedness (true for signed),
30          // endianness (true for big endian) in that order as format information.
31          AudioFormat format = new AudioFormat(sampleRate, 16, 1, true, true);
32          DataLine.Info info = new DataLine.Info(SourceDataLine.class, format);
33          SourceDataLine audioline = null;
34  
35          int bufferSize = 1000;
36          // 16-bit samples so two indices of a byte array represent a single sample.
37          // Thus, amount of samples per buffer is half the size of the array.
38          int samplesPerBuffer = bufferSize / 2;
39          byte[] sampleBuffer = new byte[bufferSize];
40          Arrays.fill(sampleBuffer, (byte) 0);
41  
42          try {
43  
44              audioline = (SourceDataLine) AudioSystem.getLine(info);
45              audioline.open(format);
46              audioline.start();
47  
48              // Game loop :)
49              int timer = 1000;
50              long lastTime = System.currentTimeMillis();
51  
52              while (timer > 0) {
53                  generateSineIntoBuffer(sampleBuffer, sampleRate, 440, samplesPerBuffer);
54                  audioline.write(sampleBuffer, 0, bufferSize);
55                  timer -= (System.currentTimeMillis() - lastTime);
56              }
57  
58          } catch (LineUnavailableException e) {
59  
60          } finally {
61  
62              audioline.drain();
63              audioline.close();
64  
65          }
66      }
67  
68      // Calculate the y value of a sine wave at frequency for each discrete point x,
69      // and put into buffer.
70      public static void generateSineIntoBuffer(byte[] sampleBuffer, int sampleRate, double frequency, int samplesPerBuffer) {
71  
72          long sampleIndex = 0;
73          long samplesInPeriod = (long) (sampleRate / frequency);
74  
75          int index = 0;
76  
77          for (int i = 0; i < samplesPerBuffer; i++) {
78  
79              double x = sampleIndex / (double) samplesInPeriod;
80              double value = Math.sin(2.0 * Math.PI * x);
81              sampleIndex = (sampleIndex + 1) % samplesInPeriod;
82  
83              double ds = value * Short.MAX_VALUE;
84              short ss = (short) Math.round(ds);
85              // Big endian, shift first eight bits and add as first part of sample
86              sampleBuffer[index++] = (byte) (ss >> 8);
87              sampleBuffer[index++] = (byte) (ss & 0xFF);
88          }
89      }
