6   package fi.impliedfeline.juisynth.logic;
7   
8   import java.util.Arrays;
9   import javax.sound.sampled.AudioFormat;
10  import javax.sound.sampled.AudioSystem;
11  import javax.sound.sampled.DataLine;
12  import javax.sound.sampled.LineUnavailableException;
13  import javax.sound.sampled.SourceDataLine;
14  
15  /**
16   * Starts a thread and writes sample data fetched from SignalSource
17   * into SourceDataLine.
18   * @author juicyp
19   */
20  public class Player extends Thread {
21  
22      private static final int SAMPLE_RATE = 88200;
23      private static final int SAMPLE_SIZE = 16;
24      private static final int CHANNELS = 1;
25      private static final boolean SIGNED = true;
26      private static final boolean BIG_ENDIAN = true;
27  
28      private static final int BUFFER_SIZE = 200;
29      // 16-bit samples so two indices of a byte array represent a single sample.
30      // Thus, amount of samples per buffer is half the size of the array.
31      private static final int SAMPLES_PER_BUFFER = BUFFER_SIZE / 2;
32  
33      private AudioFormat format;
34      private DataLine.Info info;
35      private SourceDataLine audioline;
36  
37      private boolean done;
38      private byte[] sampleBuffer = new byte[BUFFER_SIZE];
39      private SignalSource signalSource;
40      private int bufferIndex;
41      private double frequency = 440;
42  
43      public Player() {
44          // The constuctor of AudioFormat takes the sample rate, sample resolution in bits,
45          // amount of channels, signedness (true for signed),
46          // endianness (true for big endian) in that order as format information.
47          format = new AudioFormat(SAMPLE_RATE, SAMPLE_SIZE, CHANNELS, SIGNED, BIG_ENDIAN);
48          info = new DataLine.Info(SourceDataLine.class, format);
49      }
50  
51      @Override
52      public void run() {
53  
54          done = false;
55  
56          try {
57  
58              audioline = (SourceDataLine) AudioSystem.getLine(info);
59              audioline.open(format);
60              audioline.start();
61              int bufferIndex = 0;
62  
63              while (!done) {
64                  writeBuffer();
65              }
66  
67          } catch (LineUnavailableException e) {
68  
69          } finally {
70  
71              audioline.drain();
72              audioline.close();
73  
74          }
75  
76      }
77      
78      /**
79       * Starts the player on a new thread.
80       * @see stopPlayer
81       */
82      public void startPlayer() {
83          if (signalSource != null) {
84              start();
85          }
86      }
87  
88      /**
89       * Sets a flag to kill the thread.
90       * @see startPlayer
91       */
92      public void stopPlayer() {
93          done = true;
94      }
95      
96      public void setFrequency(double frequency) {
97          this.frequency = frequency;
98      }
99  
100     public void setSignalSource(SignalSource signalSource) {
101         this.signalSource = signalSource;
102     }
103 
104     private void writeBuffer() {
105 
106         Arrays.fill(sampleBuffer, (byte) 0);
107 
108         int index = 0;
109 
110         for (int i = 0; i < SAMPLES_PER_BUFFER; i++) {
111 
112             // Maybe just use same SignalStatus instance for successive sample fetches?
113             // Less memory garbage
114             SignalStatus signal = new SignalStatus(SAMPLE_RATE, bufferIndex, frequency);
115             bufferIndex++;
116             signalSource.generateSample(signal);
117 
118             double ds = signal.getAmplitude() * Short.MAX_VALUE;
119             short ss = (short) Math.round(ds);
120             // Big endian, shift first eight bits and add as first part of sample
121             sampleBuffer[index++] = (byte) (ss >> 8);
122             sampleBuffer[index++] = (byte) (ss & 0xFF);
123         }
124 
125         audioline.write(sampleBuffer, 0, BUFFER_SIZE);
126     }
127 }
